update count=2:
	git log --pretty=oneline --abrev-commit --all --graph --decorate
	pretty=oneline用一行展示信息
	abrev-commit简化每次提交的唯一标识符
	graph以图的形式表示出来
	decorate git版本过低时要打开，新版git默认开启(主要针对Mac)
update count=3:
	查看文件的命令
	ls 查看工作区的文件
	git ls-files 查看暂存区的文件
update count=4:
	git diff相关
	git diff 默认查看工作区和暂存区的不同
	git diff HEAD 比较工作区和版本库的不同
	git diff --cached 比较暂存区和版本库之间的不同
	git diff commitID commitID 比较两个版本之间的不同
	git diff commitID HEAD 比较某个版本和最新版本之间的不同
	git diff HEAD~n HEAD
	git diff HEAD^n HEAD 比较前n个版本和最新版本的不同 ~ ^两个都可
	git diff branch_name branch_name 比较两个分支之间的差异
